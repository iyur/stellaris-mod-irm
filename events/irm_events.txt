# ON_ACTION / TRIGGERED

namespace = irm

# Initialize modules
event = {
	id = irm.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				nor = {
					has_global_flag = irm_mod_enabled
					has_global_flag = irm_mod_disabled
				}
			}
			set_global_flag = irm_mod_enabled
			irm_reg_mod_change = { mod = irm_mod type = ini }
			irm_reg_mod_change = { mod = irm_mod type = upd }
		}
		if = {
			limit = {
				nor = {
					has_global_flag = irm_mod_sector_enabled
					has_global_flag = irm_mod_sector_disabled
				}
			}
			switch = {
				trigger = has_global_flag
					irm_mod_enabled = { set_global_flag = irm_mod_sector_enabled }
					irm_mod_disabled = { set_global_flag = irm_mod_sector_disabled }
			}
			irm_reg_mod_change = { mod = irm_mod_sector type = upd }
			if = {
				limit = { has_global_flag = irm_mod_sector_enabled }
				every_playable_country = {
					limit = {
						is_ai = no
						is_country_type = default
					}
					# dummy sector
					country_event = { id = irm.11 }
					# sector dates
					country_event = { id = irm_a.21 }
					# effect provider module
					every_owned_planet = {
						limit = { not = { has_deposit = irm_d_core_district } }
						planet_event = { id = irm_a.11 }
					}
				}
			}
		}
		log = "irm mod settings initialized"
	}

}

# Creates dummy sector country for each human player
country_event = {
	id = irm.11
	hide_window = yes
	mean_time_to_happen = { days = 1 }

	trigger = {
		has_global_flag = irm_mod_enabled
		is_ai = no
		is_country_type = default
		not = {
			any_country = {
				is_country_type = irm_country_sector
				has_country_flag = irm_country_sector_@root
			}
		}
	}
	immediate = {
		irm_sector_make_dummy = yes
	}

}

namespace = irm_a # (gameplay)

# Creates a core district as effect provider
planet_event = {
	id = irm_a.11
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				no_scope = { has_global_flag = irm_mod_sector_enabled }
				exists = owner
				owner = {
					is_ai = no
					is_country_type = default
				}
				not = { has_deposit = irm_d_core_district }
			}
			add_deposit = irm_d_core_district
		}
	}
}

# Removes core district
planet_event = {
	id = irm_a.12
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				has_deposit = irm_d_core_district
				or = {
					# mod disabled
					has_owner = no
					and = {
						exists = owner
						owner = { is_ai = yes }
					}
				}
			}
			remove_deposit = irm_d_core_district
		}
	}
}

# Writes the date of sector foundation
country_event = {
	id = irm_a.21
	hide_window = yes
	mean_time_to_happen = { days = 1 }

	trigger = {
		is_ai = no
		is_country_type = default
		any_owned_sector = {
			not = { has_sector_flag = irm_sector_date }
		}
	}

	immediate = {
		every_owned_sector = {
			limit = { not = { has_sector_flag = irm_sector_date } }
			# if no data available, try to pick the date from the oldest colony
			ordered_owned_planet = {
				limit = { is_colony = yes }
				position = 0
				order_by = trigger:colony_age
				prev = {
					set_variable = { which = irm_temp_date value = prev.trigger:colony_age }
					multiply_variable = { which = irm_temp_date value = -30 }
					set_saved_date = { key = irm_sector_date days_from_present = irm_temp_date }
					irm_unset = { which = irm_temp_date }
				}
			}

		}
	}
}

# Monthly pulse
namespace = irm_m

event = {
	id = irm_m.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			# mod enabled
			limit = {
				has_global_flag = irm_mod_enabled
				has_global_flag = irm_mod_sector_enabled
			}
			# players only
			every_playable_country = {
				limit = {
					is_ai = no
					is_country_type = default
				}
				# return idle
				country_event = { id = irm_m.21 days = 1 random = 5 }
				# sector agenda
				every_owned_sector = { irm_sector_update_agenda = yes }
			}
		}
	}

}

			}
		}
	}

# - Pop faction demands fake tooltips
	country_event = {
		id = irm_y2.31
		hide_window = yes
		is_triggered_only = yes

		immediate = {
			if = {
				limit = { not = { exists = event_target:irm_regionalist_sectors_governance@root } }
				create_saved_leader = {
					event_leader = yes
					immortal = yes
					hide_leader = yes
					species = owner_main_species
					key = irm_regionalist_sectors_governance
					effect = {
						save_global_event_target_as = irm_regionalist_sectors_governance@root	
					}
				}
			}
			event_target:irm_regionalist_sectors_governance@root = {
				set_name = { key = "irm.ui.x" }
				set_variable = { which = irm_counter_governance value = 0 }
				prev = {
					every_owned_sector = {
						limit = {
							not = { any_owned_planet = { is_in_frontier_space = yes } }
							not = {
								sector_capital = { exists = leader }
							}
						}
						prevprev = {
							change_variable = { which = irm_counter_governance value = 1 }
							set_name = {
								key = irm_regionalist_dem_governance_dummy1
								variable_string = "[This.GetName]" # existing
								variable_string = "[Prev.GetName]" # next sector
								variable_string = "[Prev.irm_GetSectorType]" # type
							}
						}
					}
				}
				if = {
					limit = { check_variable = { which = irm_counter_governance value > 0 } }
					set_name = {
						key = irm_regionalist_dem_governance_dummy2
						variable_string = "[Owner.irm_regionalist_sectors_governance.irm_counter_governance]"
						variable_string = "[This.GetName]"
					}
				}
			}
		}

	}

	country_event = {
		id = irm_y5.11
		hide_window = yes
		is_triggered_only = yes

		immediate = {
			every_owned_sector = {
				# criterias to laucnh a new colonization
				limit = {
					nor = {
						irm_sector_has_size_1 = yes
						has_sector_flag = irm_ui_sector_reform
						any_system_within_border = {
							any_system_planet = {
								is_under_colonization = yes
							}
						}
					}
					has_sector_flag = irm_sector_colonization_allowed
					any_system_within_border = {
						any_system_planet = {
							irm_is_colonizable = yes
							can_colonize = { who = prevprev.owner status = yes }
						}
					}
				}
				irm_sector_try_colonize = yes
			}
		}
