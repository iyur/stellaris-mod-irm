irm_regionalist = {
	election_header = "GFX_faction_header_yellow"
	use_guiding_ethic_as_pop_filter = no
	can_rename = no

	visual_ethic = "manifesti_visual_ethic"

	unique = yes
	support_multiplier = 0.5

	resources = {
	}

	extortion_resource = {
	}

	is_potential = {
		is_ai = no
		years_passed > 5
		num_sectors >= 1
		num_owned_planets >= 2
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		is_shackled_robot = no
		can_think = yes
	}

	attraction = {
		base = 100

		# modifier = {
		# 	factor = 1.25
		# 	exists = from
		# 	from = {
		# 		exists = leader
		# 		leader = { is_ruler = yes }
		# 	}
		# }

		# territorial factors
		modifier = {
			factor = 0.25
			exists = planet
			planet = {
				is_capital = yes
			}
		}
		modifier = {
			factor = 0.25
			exists = planet
			planet = {
				exists = sector
				sector = {
					irm_is_sector_core = yes
				}
			}
		}
		modifier = {
			factor = 1.25
			exists = planet
			planet = {
				exists = sector
				sector = {
					not = { irm_is_sector_core = yes }
				}
			}
		}
		modifier = {
			factor = 2.00
			exists = planet
			planet = {
				is_in_frontier_space = yes
			}
		}

	}

	leader = {
		base = 100
		modifier = {
			factor = 0
			is_event_leader = yes
		}
	}

	# frontier planets
	demand = {
		title = "irm_regionalist_dem_frontier"
		unfulfilled_title = "irm_regionalist_dem_frontier_no"
		desc = "irm_regionalist_dem_frontier_desc"

		fulfilled_effect = 0
		unfulfilled_effect = -2.5

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				not = {
					any_owned_planet = {
						is_in_frontier_space = yes
					}
				}
			}
		}
	}

	# unassigned leaders
	demand = {
		title = "irm_regionalist_dem_governance"
		unfulfilled_title = "irm_regionalist_dem_governance_no"
		desc = "irm_regionalist_dem_governance_desc"

		fulfilled_effect = 0
		unfulfilled_effect = -2.5

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				not = {
					any_owned_sector = {
						exists = sector_capital
						sector_capital = {
							not = { exists = leader }
						}
					}
				}
			}
		}
	}

	# non background leaders
	demand = {
		title = "irm_regionalist_dem_native"
		unfulfilled_title = "irm_regionalist_dem_native_no"
		desc = "irm_regionalist_dem_native_desc"

		fulfilled_effect = 0
		unfulfilled_effect = -2.5

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				num_sectors > 1
				not = {
					any_owned_sector = {
						exists = sector_capital
						sector_capital = {
							exists = leader
							leader = {
								prevprev = {
									not = {
										any_owned_planet = {
											prevprev = {
												is_background_planet = prev
											}
										}
									}
								}
							}
						}
					}
				}
			}
		}
	}

	on_create = {
		set_name = "pft_irm_regionalist"
	}

	on_destroy = {
	}

	actions = {
	}
}
